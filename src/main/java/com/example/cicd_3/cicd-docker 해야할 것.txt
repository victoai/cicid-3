# 깃헙에서 해야할  것

APPLICATION_PROPERTIES   =>작성한 application.properties  파일 내용
EC2_HOST   => EC2 IP
EC2_PRIVATE_KEY   => EC2 .pem 키
EC2_USERNAME  => ubuntu
AWS_ACCESS_KEY_ID    => ECR 사용자  id
AWS_SECRET_ACCESS_KEY    => ECR 사용자 비번
